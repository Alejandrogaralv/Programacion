// algoritmo que calcule el mayor y el menor de 3 números leído por teclado

PROGRAMA:MayorMenor

Inicio
    Entero num, mayor, menor
    Imprimir "Introduce tres números"
    para (entero i hasta 4 incremento 1)
        leer num
        si (i = 1)
            mayor<-num 
            menor<-num

        si números
            si (num > mayor)
                mayor <- num
            fin si
            si  (num< menor)
                menor <- num
            fin si
        fin si
    fin para
    imprimir...
fin
