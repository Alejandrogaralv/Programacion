PROGRAMA: OperacionesBasicas

Funcion suma (entero num1, entero num2) : entero
    entero suma
    suma<- num1 + num2
    Retornar suma
FinFunción

Funcion resta (entero num1, entero num2) : entero
    entero resta
    resta<- num1 - num2
    Retornar resta
FinFuncion

Funcion mult (entero num1, entero num2) : entero
    entero mult
    mult<- num1 * num2
    Retornar mult
FinFuncion

Funcion divi (entero num1, entero num2) : real
    entero divi
    divi<- num1 / num2
    Retornar divi
FinFuncion
